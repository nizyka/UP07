namespace DE
{
    /// <summary>
    /// Логика взаимодействия для MainWindow.xaml
    /// </summary>
    public partial class MainWindow : Window
    {
        public MainWindow()
        {
            InitializeComponent();
            MyDb();
        }

        private void MyDb()
        {
            var massiv = from prm in Entities.GetContext().ProductMaterial
                            join prod in Entities.GetContext().Product on prm.ProductID equals prod.ID
                            join mat in Entities.GetContext().Material on prm.MaterialID equals mat.ID
                            join prty in Entities.GetContext().ProductType on prod.ProductTypeID equals prty.ID
                            select new
                            {
                                prod = prod.ProductType,
                                type = prty.Title,
                                name = prod.Title,
                                art = prod.ArticleNumber,
                                mater = mat.Title,
                                sto = prm.Count

                            };
            dataGridProd.ItemsSource = massiv.ToList();
        }
