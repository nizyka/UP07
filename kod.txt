        private void dell_button(object sender, RoutedEventArgs e)
        {
            if (dataGrid.SelectedItem !=null)
            {
                int ID = ((View_1)dataGrid.SelectedItem).ID;
                Driver driver = Entities2.GetContext().Driver.Where(p => p.ID_dri == ID).FirstOrDefault();
                Entities2.GetContext().Driver.Remove(driver);
                Entities2.GetContext().SaveChanges();
                Entities2.GetContext().ChangeTracker.Entries().ToList().ForEach(p => p.Reload());
                dataGrid.ItemsSource = Entities2.GetContext().View_1.ToList();
            }
        }
    }
